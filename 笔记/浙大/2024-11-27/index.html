int cal_pri(char a){
    if(a == '+' || a == '-'){
        return 1;
    }else if(a == '*' || a == '/'){
        return 2;
    }else if(a == '('){
        return 0;
    }else if(a == ')'){
        return 3;
    }else{
        return -1;
    }
}

double calculate(double num1, double num2, char op){
    if(num2 == 0 && op == '/'){
        return INF;
    }
    switch(op){
        case '+': return (double)(num1 + num2);
        case '-': return (double)(num1 - num2);
        case '*': return (double)(num1 * num2);
        case '/': return (double)(num1 / num2);
    }
}

    return -1;
}
